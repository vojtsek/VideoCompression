636  bool operator==(const GenericValue& rhs) const {
637  if (GetType() != rhs.GetType())
638  return false;
639 
640  switch (GetType()) {
641  case kObjectType: // Warning: O(n^2) inner-loop
642  if (data_.o.size != rhs.data_.o.size)
643  return false;
644  for (ConstMemberIterator lhsMemberItr = MemberBegin(); lhsMemberItr != MemberEnd(); ++lhsMemberItr) {
645  ConstMemberIterator rhsMemberItr = rhs.FindMember(lhsMemberItr->name);
646  if (rhsMemberItr == rhs.MemberEnd() || lhsMemberItr->value != rhsMemberItr->value)
647  return false;
648  }
649  return true;
650 
651  case kArrayType:
652  if (data_.a.size != rhs.data_.a.size)
653  return false;
654  for (SizeType i = 0; i < data_.a.size; i++)
655  if ((*this)[i] != rhs[i])
656  return false;
657  return true;
658 
659  case kStringType:
660  return StringEqual(rhs);
661 
662  case kNumberType:
663  if (IsDouble() || rhs.GetDouble())
664  return GetDouble() == rhs.GetDouble(); // May convert one operand from integer to double.
665  else
666  return data_.n.u64 == rhs.data_.n.u64;
667 
668  default: // kTrueType, kFalseType, kNullType
669  return true;
670  }
